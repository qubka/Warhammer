chaos_imperial_knight = {
	damage = 4.0
	health = 6.0
	morale = 4.0
	collateral_damage = 1.00
	war_exhaustion = 0.0
	time = 500
	
	icon_frame = 35
	
	resources = {
		category = armies
		cost = {
			minerals = 1050
		}

		upkeep = {
			energy = 6.5
		}
	}
	
	#prerequisites = { "tech_titans_army" }
	
	allow = {
		#has_building = chaos_building_imperial_knight_base_1
	
		owner = {
			NOT = { 
				check_variable = {
					which = "chaos_imperial_knight_count" 
					value = 15
				}
			}
		}
		
	}
	
	on_queued = {
		owner = {
			change_variable = {
				which = "chaos_imperial_knight_count" 
				value = 1
			}
		}
	}
	
	on_unqueued = {
		owner = {
			change_variable = {
				which = "chaos_imperial_knight_count" 
				value = -1
			}
		}
	}

	potential = {
		planet = {
			#has_building = chaos_building_imperial_knight_base_1
		}
		owner = { has_country_flag = chaos_empire }
	}
}
