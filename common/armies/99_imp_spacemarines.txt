dark_angels = {
	damage = 3.0
	health = 8.0
	morale = 3.0
	collateral_damage = 1.00
	war_exhaustion = 0.0
	time = 360
	
	icon_frame = 10
	
	resources = {
		category = armies
		cost = {
			military_goods = 35
			gene_seed = 10
		}

		upkeep = {
			energy = 3
		}
	}

	allow = {
		has_building = building_chapel_barracks
		
		owner = {
			NOT = { 
				check_variable = {
					which = "dark_angels_count" 
					value = 6
				}
			}
		}
	}
	
	on_queued = {
		owner = {
			change_variable = {
				which = "dark_angels_count" 
				value = 1
			}
		}
	}
	
	on_unqueued = {
		owner = {
			change_variable = {
				which = "dark_angels_count" 
				value = -1
			}
		}
	}
	
	potential = {
		OR = {
			owner = { has_country_flag = the_dark_angels }
			AND = {
				planet = { has_planet_flag = caliban_planet }
				owner = { has_country_flag = human_empire }
			}
		}
	}
}
