#pc_hive_medium
#class = "pc_hive_medium"
#flags = { hiveplanet_planet }
event = {
    id = planets_and_stuf_setup.1
    hide_window = yes
    fire_only_once = yes
    
	immediate = {
		every_planet = {
			limit = { has_owner = yes is_planet_class = pc_hive_medium }
			add_modifier = { modifier = "hive_planet" days = -1 }
			add_building = building_system_capital
			while = {
			   count = 2
			   add_building = energy_factory
			}
			while = {
			   count = 2
			   add_building = building_foundry_1
			}
			while = {
				count = 20
				add_district_and_planet_size_if_needed_effect = {
					district = district_arcology_housing
				}
			}
			while = {
				count = 2
				add_district_and_planet_size_if_needed_effect = {
					district = district_arcology_civilian_industry
				}
			}
			while = {
				count = 3
				add_district_and_planet_size_if_needed_effect = {
					district = district_arcology_leisure
				}
			}
			while = {
				count = 50
				create_pop = {
					species = owner_main_species
				}
			}
		}
	}
}
